    #include 
      using namespace std;
      
      int main() {
          double salario, valorItem;
          int numParcelas;
      
          cout << "Digite o salário do cliente: R$ ";
          cin >> salario;
          cout << "Digite o valor do item: R$ ";
          cin >> valorItem;
          cout << "Digite o número de parcelas desejado: ";
          cin >> numParcelas;
      
          double valorParcela = valorItem / numParcelas;
      
          bool autorizado = true;
          string motivo;
      
          if (numParcelas > 12) {
              autorizado = false;
              motivo = "Número de parcelas superior a 12.";
          } else {
              double valorParcelaComJuros = valorParcela * 1.05; // Valor da parcela com juros de 5%
              if (valorParcelaComJuros > (salario * 0.3)) {
                  autorizado = false;
                  motivo = "Valor da parcela superior a 30% do salário.";
              }
          }
      
          switch (autorizado) {
              case true:
                  cout << "Pagamento autorizado." << endl;
                  break;
              case false:
                  cout << "Pagamento não autorizado. Motivo: " << motivo << endl;
                  break;
          }
      
          return 0;
      }
      
    

  
    10)Usando if para calcular o valor da conta de energia:
    #include 
      using namespace std;
      
      int main() {
          double consumoKWh;
          char bandeira;
      
          cout << "Digite o consumo em KW/h: ";
          cin >> consumoKWh;
          cout << "Digite a bandeira (V para verde, A para amarela, R para vermelha): ";
          cin >> bandeira;
      
          double custo = consumoKWh * 2.0;
      
          if (bandeira == 'R') {
              custo += custo * 0.055; // Acréscimo de 5,5% para bandeira vermelha
          } else if (bandeira == 'A') {
              custo += custo * 0.025; // Acréscimo de 2,5% para bandeira amarela
          }
      
          cout << "Valor total da fatura: R$ " << custo << endl;
      
          return 0;
      }

      using namespace std;
      
      int main() {
          double consumoKWh;
          char bandeira;
      
          cout << "Digite o consumo em KW/h: ";
          cin >> consumoKWh;
          cout << "Digite a bandeira (V para verde, A para amarela, R para vermelha): ";
          cin >> bandeira;
      
          double custo = consumoKWh * 2.0;
      
          switch (bandeira) {
              case 'R':
                  custo += custo * 0.055; // Acréscimo de 5,5% para bandeira vermelha
                  break;
              case 'A':
                  custo += custo * 0.025; // Acréscimo de 2,5% para bandeira amarela
                  break;
              case 'V':
                  // Não há acréscimo para bandeira verde
                  break;
              default:
                  cout << "Bandeira inválida." << endl;
                  return 1;
          }
      
          cout << "Valor total da fatura: R$ " << custo << endl;
      
          return 0;
      }
